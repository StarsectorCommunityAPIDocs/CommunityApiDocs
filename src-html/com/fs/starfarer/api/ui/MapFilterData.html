001package com.fs.starfarer.api.ui;
002
003public class MapFilterData {
004        public boolean starscape;
005        public boolean names = true;
006        public boolean constellations;
007        public boolean factions;
008        public boolean missions;
009        public boolean fuel;
010        public boolean exploration;
011        public boolean legend = true;
012        public float fuelColorAlphaMult = 1f;
013        public float fuelRangeMult = 1f;
014        
015        public MapFilterData(boolean forNewGame) {
016                super();
017                
018                if (forNewGame) {
019                        starscape = true;
020                        names = true;
021                        factions = true;
022                        
023                        missions = false;
024                        fuel = false;
025                        exploration = false;
026                        legend = true;
027                }
028        }
029        
030}
