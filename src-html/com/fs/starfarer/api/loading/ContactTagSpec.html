001package com.fs.starfarer.api.loading;
002
003import java.awt.Color;
004
005import org.json.JSONObject;
006
007public class ContactTagSpec {
008        
009//      "name":"Important",
010//      "color":"buttonShortcut",
011//      "width":120,
012//      "putFirst":true,
013//      "sort":10,
014
015        private String tag;
016        private String name;
017        private Color color;
018        private float width;
019        private boolean putFirst;
020        private float sort;
021        public ContactTagSpec(String tag, Color color, JSONObject json) {
022                this.tag = tag;
023                this.color = color;
024                
025                name = json.optString("name", null);
026                width = (float) json.optDouble("width", 0);
027                putFirst = json.optBoolean("putFirst", false);
028                sort = (float) json.optDouble("sort", 0);
029        }
030        
031        
032        public String getTag() {
033                return tag;
034        }
035        public void setTag(String tag) {
036                this.tag = tag;
037        }
038        public String getName() {
039                return name;
040        }
041        public void setName(String name) {
042                this.name = name;
043        }
044        public Color getColor() {
045                return color;
046        }
047        public void setColor(Color color) {
048                this.color = color;
049        }
050        public float getWidth() {
051                return width;
052        }
053        public void setWidth(float width) {
054                this.width = width;
055        }
056        public boolean isPutFirst() {
057                return putFirst;
058        }
059        public void setPutFirst(boolean putFirst) {
060                this.putFirst = putFirst;
061        }
062        public float getSort() {
063                return sort;
064        }
065        public void setSort(float sort) {
066                this.sort = sort;
067        }
068        
069
070        
071        
072}
