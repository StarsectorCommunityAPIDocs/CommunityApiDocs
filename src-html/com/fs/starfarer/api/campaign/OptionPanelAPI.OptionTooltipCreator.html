001package com.fs.starfarer.api.campaign;
002
003import java.awt.Color;
004import java.util.List;
005
006import com.fs.starfarer.api.impl.campaign.rulecmd.SetStoryOption.StoryOptionParams;
007import com.fs.starfarer.api.ui.TooltipMakerAPI;
008import com.fs.starfarer.api.ui.ValueDisplayMode;
009
010
011public interface OptionPanelAPI {
012        
013        public static interface OptionTooltipCreator {
014                void createTooltip(TooltipMakerAPI tooltip, boolean hadOtherText);
015        }
016        
017        
018        void setTooltipHighlights(Object data, String ... highlights);
019        void setTooltipHighlightColors(Object data, Color ... colors);
020        
021        void clearOptions();
022        void addOption(String text, Object data);
023        void addOption(String text, Object data, String tooltip);
024        void addOption(String text, Object data, Color color, String tooltip);
025        
026        /**
027         * Sets an alternate shortcut that works in addition to the number key.
028         * @param data
029         * @param code constant from org.lwjgl.input.Keyboard
030         * @param ctrl whether Control needs to be down to trigger this shortcut.
031         * @param alt whether Alt needs to be down to trigger this shortcut.
032         * @param shift whether Shift needs to be down to trigger this shortcut.
033         * @param putLast ignored
034         */
035        void setShortcut(Object data, int code, boolean ctrl, boolean alt, boolean shift, boolean putLast);
036        
037        /**
038         * Only works for options, not selectors.
039         * @param data
040         * @param enabled
041         */
042        void setEnabled(Object data, boolean enabled);
043        
044        
045        void setTooltip(Object data, String tooltipText);
046        
047        /**
048         * A user-adjustable bar useful for picking a value from a range.
049         * @param text Text to show above the bar.
050         * @param data ID of the bar, used to get/set its state.
051         * @param color Bar color.
052         * @param width Width in pixels, including value label on the right.
053         * @param maxValueWidth Width of the value label on the right.
054         * @param minValue Minimum value (when bar is all the way to the left).
055         * @param maxValue Maximum value (bar all the way to the right).
056         * @param mode How to display the value - as a percentage, X/Y, etc.
057         * @param tooltip Tooltip text. Can be null.
058         */
059        void addSelector(String text, Object data, Color color,
060                                         float width, float maxValueWidth, float minValue, float maxValue,
061                                         ValueDisplayMode mode, String tooltip);
062        
063        boolean hasSelector(Object data);
064        
065        void setSelectorValue(Object data, float value);
066        
067        float getSelectorValue(Object data);
068        float getMinSelectorValue(Object data);
069        float getMaxSelectorValue(Object data);
070        boolean hasOptions();
071        
072        List getSavedOptionList();
073        void restoreSavedOptions(List list);
074        
075        void addOptionConfirmation(Object optionId, String text, String yes, String no);
076        boolean hasOption(Object data);
077        void addOptionConfirmation(Object data, StoryPointActionDelegate confirmDelegate);
078        void addOptionTooltipAppender(Object data, OptionTooltipCreator optionTooltipCreator);
079        void setOptionText(String text, Object data);
080        boolean hasOptionTooltipAppender(Object data);
081        boolean optionHasConfirmDelegate(Object data);
082        Object getOptionDataBeingConfirmed();
083        void removeOption(Object data);
084        
085        void setStoryOptionParams(Object data, StoryOptionParams params, StoryPointActionDelegate delegate);
086}
