001package com.fs.starfarer.api.campaign.listeners;
002
003import java.awt.Color;
004import java.util.List;
005
006import com.fs.starfarer.api.campaign.econ.Industry;
007import com.fs.starfarer.api.campaign.econ.Industry.IndustryTooltipMode;
008import com.fs.starfarer.api.ui.TooltipMakerAPI;
009import com.fs.starfarer.api.util.Misc;
010
011public interface IndustryOptionProvider {
012
013        public class IndustryOptionData {
014                public String text;
015                public Color color;
016                public boolean enabled = true;
017                public Object id;
018                public Industry ind;
019                public IndustryOptionProvider provider;
020                
021                
022                public IndustryOptionData(String text, Object id, Industry ind, IndustryOptionProvider provider) {
023                        this.text = text;
024                        this.id = id;
025                        this.ind = ind;
026                        this.provider = provider;
027                        
028                        if (ind.getMarket() != null) {
029                                color = ind.getMarket().getFaction().getBaseUIColor();
030                        } else {
031                                color = Misc.getBasePlayerColor();
032                        }
033                }
034        }
035        
036        public List<IndustryOptionData> getIndustryOptions(Industry ind);
037        public void createTooltip(IndustryOptionData opt, TooltipMakerAPI tooltip, float width);
038        
039        public void optionSelected(IndustryOptionData opt, DialogCreatorUI ui);
040        public void addToIndustryTooltip(Industry ind, IndustryTooltipMode mode, TooltipMakerAPI tooltip, float width, boolean expanded);
041}
