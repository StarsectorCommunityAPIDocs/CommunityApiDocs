001package com.fs.starfarer.api.impl.campaign.skills;
002
003import com.fs.starfarer.api.campaign.FleetDataAPI;
004import com.fs.starfarer.api.characters.FleetTotalItem;
005import com.fs.starfarer.api.characters.FleetTotalSource;
006import com.fs.starfarer.api.characters.MutableCharacterStatsAPI;
007import com.fs.starfarer.api.characters.ShipSkillEffect;
008import com.fs.starfarer.api.characters.SkillSpecAPI;
009import com.fs.starfarer.api.combat.MutableShipStatsAPI;
010import com.fs.starfarer.api.combat.ShipAPI;
011import com.fs.starfarer.api.combat.ShipAPI.HullSize;
012import com.fs.starfarer.api.fleet.FleetMemberAPI;
013import com.fs.starfarer.api.impl.campaign.ids.Strings;
014import com.fs.starfarer.api.ui.TooltipMakerAPI;
015import com.fs.starfarer.api.util.Misc;
016
017public class CarrierGroup {
018        
019        public static float REPLACEMENT_RATE_PERCENT = 50;
020        
021        public static float OFFICER_MULT = 1.5f;
022        public static boolean isOfficer(MutableShipStatsAPI stats) {
023                if (stats.getEntity() instanceof ShipAPI) {
024                        ShipAPI ship = (ShipAPI) stats.getEntity();
025                        if (ship == null) return false;
026                        return !ship.getCaptain().isDefault();
027                } else {
028                        FleetMemberAPI member = stats.getFleetMember();
029                        if (member == null) return false;
030                        return !member.getCaptain().isDefault();
031                }
032        }
033        public static class Level1 extends BaseSkillEffectDescription implements ShipSkillEffect, FleetTotalSource {
034                
035                public FleetTotalItem getFleetTotalItem() {
036                        return getFighterBaysTotal();
037                }
038                
039                public void apply(MutableShipStatsAPI stats, HullSize hullSize, String id, float level) {
040                        if (hasFighterBays(stats)) {
041                                float rateBonus = computeAndCacheThresholdBonus(stats, "cg_rep_rate", REPLACEMENT_RATE_PERCENT, ThresholdBonusType.FIGHTER_BAYS);
042                                if (isOfficer(stats)) rateBonus *= OFFICER_MULT;
043                                float timeMult = 1f / ((100f + rateBonus) / 100f);
044                                stats.getFighterRefitTimeMult().modifyMult(id, timeMult);
045                        }
046                }
047                
048                public void unapply(MutableShipStatsAPI stats, HullSize hullSize, String id) {
049                        stats.getFighterRefitTimeMult().unmodifyMult(id);
050                }
051                
052                public String getEffectDescription(float level) {
053                        return null;
054                }
055                        
056                public void createCustomDescription(MutableCharacterStatsAPI stats, SkillSpecAPI skill, 
057                                                                                        TooltipMakerAPI info, float width) {
058                        init(stats, skill);
059                        
060                        FleetDataAPI data = getFleetData(null);
061                        float rateBonus = computeAndCacheThresholdBonus(data, stats, "cg_rep_rate", REPLACEMENT_RATE_PERCENT, ThresholdBonusType.FIGHTER_BAYS);
062                        
063                        info.addPara("+%s faster fighter replacement rate (maximum: %s)", 0f, hc, hc,
064                                        "" + (int) rateBonus + "%",
065                                        "" + (int) REPLACEMENT_RATE_PERCENT + "%");
066                        addFighterBayThresholdInfo(info, data);
067                        info.addPara(indent + "Effect increased by %s for ships with officers, including flagship",
068                                        0f, tc, hc, 
069                                        "" + Misc.getRoundedValueMaxOneAfterDecimal(OFFICER_MULT) + Strings.X);
070                }
071                
072                public ScopeDescription getScopeDescription() {
073                        return ScopeDescription.ALL_CARRIERS;
074                }
075
076        }
077        
078
079
080}
