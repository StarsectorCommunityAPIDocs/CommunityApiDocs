001package com.fs.starfarer.api.impl.campaign.intel.raid;
002
003import java.awt.Color;
004
005import com.fs.starfarer.api.campaign.econ.MarketAPI;
006import com.fs.starfarer.api.impl.campaign.ids.MemFlags;
007import com.fs.starfarer.api.impl.campaign.intel.raid.RaidIntel.RaidStageStatus;
008import com.fs.starfarer.api.ui.TooltipMakerAPI;
009import com.fs.starfarer.api.util.IntervalUtil;
010import com.fs.starfarer.api.util.Misc;
011
012public class OrganizeStage extends BaseRaidStage {
013        
014        protected MarketAPI market;
015        protected IntervalUtil interval = new IntervalUtil(0.1f, 0.2f);
016        protected boolean wasMilitary = false;
017        
018        public OrganizeStage(RaidIntel raid, MarketAPI market, float durDays) {
019                super(raid);
020                this.market = market;
021                this.maxDays = durDays;
022                
023                wasMilitary = market.getMemoryWithoutUpdate().getBoolean(MemFlags.MARKET_MILITARY);
024        }
025        
026        
027        public void advance(float amount) {
028                if (status == RaidStageStatus.ONGOING && 
029                                (!market.isInEconomy() || (!market.getMemoryWithoutUpdate().getBoolean(MemFlags.MARKET_MILITARY) && wasMilitary))) {
030                        abort();
031                        return;
032                }
033                super.advance(amount);
034        }
035        
036        protected void updateStatus() {
037                if (maxDays <= elapsed) {
038                        status = RaidStageStatus.SUCCESS;
039                }
040        }
041        
042        public void abort() {
043                status = RaidStageStatus.FAILURE;
044        }
045        
046        public MarketAPI getMarket() {
047                return market;
048        }
049
050        public void showStageInfo(TooltipMakerAPI info) {
051                int curr = intel.getCurrentStage();
052                int index = intel.getStageIndex(this);
053                
054                Color h = Misc.getHighlightColor();
055                Color g = Misc.getGrayColor();
056                Color tc = Misc.getTextColor();
057                float pad = 3f;
058                float opad = 10f;
059                
060                int days = Math.round(maxDays - elapsed);
061                String strDays = RaidIntel.getDaysString(days);
062                
063                String timing = getForcesString() + " should begin assembling in %s " + strDays + ".";
064                if (days < 2) {
065                        timing = getForcesString() + " should begin assembling shortly.";
066                }
067                
068                String raid = getRaidString();
069                if (status == RaidStageStatus.FAILURE) {
070                        info.addPara("The " + raid + " has been disrupted in the planning stages and will not happen.", opad);
071                } else if (curr == index) {
072                        boolean known = !market.isHidden() || !market.getPrimaryEntity().isDiscoverable();
073                        if (known) {
074                                info.addPara("The " + raid + " is currently being planned " + 
075                                                market.getOnOrAt() + " " + market.getName() + ". " + timing,
076                                                opad, h, "" + days);
077                        } else {
078                                info.addPara("The " + raid + " is currently in the planning stages. " + timing,
079                                                opad, h, "" + days);
080                        }
081                }
082        }
083        
084        protected String getForcesString() {
085                return "The raiding forces";
086        }
087        
088        protected String getRaidString() {
089                return "raid";
090        }
091}
