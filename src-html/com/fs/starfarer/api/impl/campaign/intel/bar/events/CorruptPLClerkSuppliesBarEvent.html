023                if (!super.shouldShowAtMarket(market)) return false;
024                regen(market);
025                
026                if (!market.getFactionId().equals(Factions.PERSEAN)) {
027                        return false;
028                }
029                if (market.getId().equals("kazeron")) return false;
030                if (market.getStabilityValue() > 7) return false;

032                if (!market.hasSpaceport()) return false;

037//                              hasStation = true;
038//                              break;
039//                      }
040//              }
041//              if (!hasStation) return false;
042                
043                
044                return true;
045        }

080        @Override
081        protected int computeQuantity() {
082                int quantity = 30 + 10 * random.nextInt(4);
083                
084                CommodityOnMarketAPI com = market.getCommodityData(commodity);
085                int size = com.getAvailable();
086                if (size < 1) size = 1;
087                quantity *= Math.max(1, BaseIndustry.getSizeMult(size) - 2);
088                return quantity;
089        }
