003import java.awt.Color;
004import java.util.List;
005
006import com.fs.starfarer.api.Global;

017
018public class HegemonyAICoresActivityCause extends BaseHostileActivityCause2 {
019
020        public static int IGNORE_COLONY_THRESHOLD = 3;
021        
022        
023        public HegemonyAICoresActivityCause(HostileActivityEventIntel intel) {
024                super(intel);
025        }
026        
027        @Override
028        public TooltipCreator getTooltip() {
029                return new BaseFactorTooltip() {
030                        public void createTooltip(TooltipMakerAPI tooltip, boolean expanded, Object tooltipParam) {
031                                float opad = 10f;
032                                tooltip.addPara("The Hegemony considers the use of %s illegal, though it is unlikely "
033                                                + "to take much notice of what goes on at colonies of size %s or smaller.", 0f,
034                                                Misc.getHighlightColor(), "AI cores", "" + IGNORE_COLONY_THRESHOLD);
035                                
036                                if (isNegatedByPLMembership()) {
037                                        Color c = Global.getSector().getFaction(Factions.PERSEAN).getBaseUIColor();
038                                        LabelAPI label = tooltip.addPara("However, your membership in the Persean League makes it politically "
039                                                        + "difficult for the Hegemony to pursue the matter.", opad);
040                                        label.setHighlight("Persean League", "politically difficult");
041                                        label.setHighlightColors(c, Misc.getPositiveHighlightColor());
042                                }
043                        }
044                };
045        }
046        
047        public boolean isNegatedByPLMembership() {
048                //if (true) return true;
049                return PerseanLeagueMembership.isLeagueMember() && getProgress(false) > 0;
050        }
051        
052        @Override
053        public String getProgressStr() {
054                if (isNegatedByPLMembership()) return EventFactor.NEGATED_FACTOR_PROGRESS;
055                return super.getProgressStr();
056        }
057
058        @Override
059        public Color getProgressColor(BaseEventIntel intel) {
060                if (isNegatedByPLMembership()) return Misc.getPositiveHighlightColor();
061                return super.getProgressColor(intel);
062        }
063        
064        @Override
065        public boolean shouldShow() {
066                return getProgress() != 0 || isNegatedByPLMembership();
067        }
068
069        public int getProgress() {
070                return getProgress(true);
071        }
072        public int getProgress(boolean checkNegated) {
073                if (HegemonyHostileActivityFactor.isPlayerDefeatedHegemony()) return 0;
074                if (checkNegated && isNegatedByPLMembership()) return 0;
075                
076                int progress = (int) Math.round(getTotalAICorePoints());
077                
078                float unit = Global.getSettings().getFloat("hegemonyProgressUnit");
079                float mult = Global.getSettings().getFloat("hegemonyProgressMult");
080                //progress = 200;
081                int rem = progress;
082                float adjusted = 0;
083                while (rem > unit) {
084                        adjusted += unit;
085                        rem -= unit;
086                        rem *= mult;
087                }
088                adjusted += rem;
089                
090                int reduced = Math.round(adjusted);
091                if (progress > 0 && reduced < 1) reduced = 1;
092                progress = reduced;
093                
094//              int reduced = (int) Math.round(Math.pow(progress, 0.8f));
095//              if (progress > 0 && reduced <= 0) reduced = 1;
096//              progress = reduced;
097                
098                return progress;
099        }
100        
101        public String getDesc() {
102                return "AI core use";
103        }
104        
105        public float getTotalAICorePoints() {
106                float total = 0f;
107                for (StarSystemAPI system : Misc.getPlayerSystems(false)) {
108                        total += getAICorePoints(system);
109                }
110                return total;
111        }
112        
113        public static float getAICorePoints(StarSystemAPI system) {
114                float total = 0f;
115                List<MarketAPI> markets = Misc.getMarketsInLocation(system, Factions.PLAYER);
116                for (MarketAPI market : markets) {
117                        if (market.getSize() <= IGNORE_COLONY_THRESHOLD) continue;
118                        float interest = getAICorePoints(market);
119                        total += interest;
120                }
121                return total;
122        }
123        
124        public static float getAICorePoints(MarketAPI market) {
125                float total = 0f;
126                
127                float admin = Global.getSettings().getFloat("hegemonyPointsAdmin");
128                float alpha = Global.getSettings().getFloat("hegemonyPointsAlpha");
129                float beta = Global.getSettings().getFloat("hegemonyPointsBeta");
130                float gamma = Global.getSettings().getFloat("hegemonyPointsGamma");
131                
132                String aiCoreId = market.getAdmin().getAICoreId();
133                if (aiCoreId != null) {
134                        total += admin;
135                }
136                
137                for (Industry ind : market.getIndustries()) {
138                        String core = ind.getAICoreId();
139                        if (Commodities.ALPHA_CORE.equals(core)) {
140                                total += alpha;
141                        } else if (Commodities.BETA_CORE.equals(core)) {
142                                total += beta;
143                        } else if (Commodities.GAMMA_CORE.equals(core)) {
144                                total += gamma;
145                        }
146                }
147                
148                return total;
149        }
150        
151        public float getMagnitudeContribution(StarSystemAPI system) {
152                if (HegemonyHostileActivityFactor.isPlayerDefeatedHegemony()) return 0f;
153                if (isNegatedByPLMembership()) return 0f;
154                
155                if (getProgress() <= 0) return 0f;
156                
157                List<MarketAPI> markets = Misc.getMarketsInLocation(system, Factions.PLAYER);
158                
159                float perAICorePoint = Global.getSettings().getFloat("hegemonyPerAICorePoint");
160
161                float total = 0f;
162                for (MarketAPI market : markets) {
163                        float points = getAICorePoints(market);
164                        total += points;
165                }
166                
167                total = Math.round(total * 100f) / 100f;
168                
169                return total;
170        }
