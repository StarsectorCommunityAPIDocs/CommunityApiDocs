026        public boolean shouldShowAtMarket(MarketAPI market) {
027                if (!super.shouldShowAtMarket(market)) return false;
028                
029                if (market.getFactionId().equals(Factions.LUDDIC_CHURCH) ||
030                                market.getFactionId().equals(Factions.LUDDIC_PATH)) {
031                        return false;
032                }
033                
034                return true;
035        }

048        @Override
049        protected void doConfirmActionsPreAcceptText() {
050                // spawn entity with AI core etc here
051                
052                CargoAPI cargo = Global.getSector().getPlayerFleet().getCargo();
053                TextPanelAPI text = dialog.getTextPanel();
054                
055                WeightedRandomPicker<StarSystemAPI> picker = new WeightedRandomPicker<StarSystemAPI>(random);
056                for (StarSystemAPI system : Global.getSector().getStarSystems()) {
057                        if (system.hasTag(Tags.THEME_CORE_POPULATED)) continue;
058                        if (system.hasTag(Tags.THEME_REMNANT_MAIN)) continue;
059                        if (system.hasTag(Tags.THEME_REMNANT_RESURGENT)) continue;
060                        if (system.hasTag(Tags.THEME_HIDDEN)) continue;
061                        
062//                      if (system.getStar() != null && system.getStar().getSpec().isBlackHole()) continue;
063//                      if (system.getSecondary() != null && system.getSecondary().getSpec().isBlackHole()) continue;
064//                      if (system.getTertiary() != null && system.getTertiary().getSpec().isBlackHole()) continue;
065                        
066                        float sinceVisit = Global.getSector().getClock().getElapsedDaysSince(system.getLastPlayerVisitTimestamp());
067                        if (sinceVisit < 60) continue;
068                        
069                        //if (!Misc.getMarketsInLocation(system).isEmpty()) continue;
070                        
071                        picker.add(system);
072                }
073                
074                StarSystemAPI system = picker.pick();
075                if (system == null) {
076                        doDataFail();
077                        return;
078                }
079
080                //EntityLocation loc = BaseThemeGenerator.pickHiddenLocation(random, system, 100f, null);
081                EntityLocation loc = BaseThemeGenerator.pickHiddenLocationNotNearStar(random, system, 100f, null);
082                if (loc == null) {
083                        doDataFail();
084                        return;
085                }
086                
087                //AddedEntity added = BaseThemeGenerator.addNonSalvageEntity(system, loc, Entities.STABLE_LOCATION, Factions.NEUTRAL);
088                AddedEntity added = BaseThemeGenerator.addEntity(random, system, loc, Entities.TECHNOLOGY_CACHE, Factions.NEUTRAL);
089                
090                if (added == null || added.entity == null) {
091                        doDataFail();
092                        return;
093                }
094                //added.entity.setName("Technology Cache");
095                
096                //added.entity.removeTag(Tags.SALVAGEABLE);
097                added.entity.setDiscoverable(null);
098                added.entity.setDiscoveryXP(null);
099                added.entity.setSensorProfile(null);
100                
101                added.entity.addTag(Tags.EXPIRES); // so it doesn't get targeted by "analyze entity" missions
102                
103                ScientistAICoreIntel intel = new ScientistAICoreIntel(added.entity, this);
104                //intel.setImportant(true);
105                Global.getSector().getIntelManager().addIntel(intel, false, text);
106        }
