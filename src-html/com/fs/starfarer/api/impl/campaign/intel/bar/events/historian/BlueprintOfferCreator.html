001package com.fs.starfarer.api.impl.campaign.intel.bar.events.historian;
002
003import java.util.HashSet;
004import java.util.List;
005import java.util.Random;
006import java.util.Set;
007
008import com.fs.starfarer.api.campaign.PlanetAPI;
009import com.fs.starfarer.api.campaign.SectorEntityToken;
010import com.fs.starfarer.api.impl.campaign.intel.bar.events.historian.HistorianData.HistorianOffer;
011import com.fs.starfarer.api.util.WeightedRandomPicker;
012
013public abstract class BlueprintOfferCreator extends BaseHistorianOfferCreator {
014
015        public BlueprintOfferCreator(float frequency) {
016                setFrequency(frequency);
017        }
018        
019        public String getOfferId(BaseHistorianOffer offer) {
020                return null;
021        }
022        
023        public String getBlueprintIdFromOfferId(String offerId) {
024                return null;
025        }
026        
027        @Override
028        public HistorianOffer createOffer(Random random, List<HistorianOffer> soFar) {
029                HistorianData hd = HistorianData.getInstance();
030                
031                SectorEntityToken entity = pickEntity(random, false);
032                if (entity == null || random.nextFloat() < getProbabilityRuins()) {
033                        PlanetAPI planet = pickUnexploredRuins(random);
034                        if (planet != null) entity = planet;
035                }
036                if (entity == null) {
037                        entity = createEntity(random);
038                }
039                
040                if (entity == null) return null;
041                
042                WeightedRandomPicker<String> picker = new WeightedRandomPicker<String>(random);
043
044                Set<String> already = new HashSet<String>();
045                for (HistorianOffer offer : soFar) {
046                        String id = getAlreadyUsedIdFromOffer(offer);
047                        if (id != null) already.add(id);
048                }
049                
050                for (String id : hd.getGivenOffers()) {
051                        String bpId = getBlueprintIdFromOfferId(id);
052                        if (bpId != null) {
053                                already.add(bpId);
054                        }
055                }
056                
057                for (Object spec : getAllSpecs()) {
058                        String id = getIdForSpec(spec);
059                        if (already.contains(id)) continue;
060                        if (playerKnowsSpecAlready(id)) continue;
061                        
062                        float w = hd.getWeightForTags(getTagsForSpec(spec));
063                        if (w <= 0) continue;
064                        
065                        w *= getRarityForSpec(spec);
066                        picker.add(id, w);
067                }
068                
069                String data = picker.pick();
070                if (data == null) return null;
071                
072                HistorianOffer offer = createOffer(entity, data); 
073                return offer;
074        }
075        
076        protected abstract BaseHistorianOfferWithLocation createOffer(SectorEntityToken entity, String data);
077        
078        protected abstract String getAlreadyUsedIdFromOffer(HistorianOffer offer);
079        protected abstract List<Object> getAllSpecs();
080        protected abstract Set<String> getTagsForSpec(Object spec);
081        protected abstract String getIdForSpec(Object spec);
082        protected abstract boolean playerKnowsSpecAlready(String id);
083        
084        protected float getRarityForSpec(Object spec) {
085                return 1f;
086        }
087        
088        protected float getProbabilityRuins() {
089                return 0.33f;
090        }
091
092}
