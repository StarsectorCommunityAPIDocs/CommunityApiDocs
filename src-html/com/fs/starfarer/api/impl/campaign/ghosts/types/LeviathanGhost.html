001package com.fs.starfarer.api.impl.campaign.ghosts.types;
002
003import org.lwjgl.util.vector.Vector2f;
004
005import com.fs.starfarer.api.Global;
006import com.fs.starfarer.api.campaign.CampaignFleetAPI;
007import com.fs.starfarer.api.impl.campaign.ghosts.BaseSensorGhost;
008import com.fs.starfarer.api.impl.campaign.ghosts.GBGoInDirectionWithWobble;
009import com.fs.starfarer.api.impl.campaign.ghosts.GBIGenerateSlipstream;
010import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhostManager;
011import com.fs.starfarer.api.impl.campaign.ids.Tags;
012import com.fs.starfarer.api.util.Misc;
013
014public class LeviathanGhost extends BaseSensorGhost {
015
016        public LeviathanGhost(SensorGhostManager manager, int burnMod) {
017                super(manager, 0);
018                
019                CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
020                
021                float size = genFloat(400f, 500f);
022                initEntity(genHugeSensorProfile(), size);
023                setDespawnRange(-200f);
024                entity.addTag(Tags.UNAFFECTED_BY_SLIPSTREAM);
025                
026                float dirToCore = Misc.getAngleInDegrees(pf.getLocation());
027                float travelDirMin = dirToCore + 90f - 20f;
028                float travelDirMax = dirToCore + 90f + 20f;
029                float travelDir = travelDirMin + (travelDirMax - travelDirMin) * getRandom().nextFloat();
030                travelDir = Misc.normalizeAngle(travelDir);
031
032                float spawnDist = genFloat(6000, 8000);
033                float spawnAngle = (travelDir + 30f - 60f * getRandom().nextFloat()) + 180f;
034                
035                if (getRandom().nextBoolean()) {
036                        travelDir += 180f;
037                        spawnAngle += 180f;
038                }
039                
040                Vector2f loc = Misc.getUnitVectorAtDegreeAngle(spawnAngle);
041                loc.scale(spawnDist);
042                Vector2f.add(loc, pf.getLocation(), loc);
043                setLoc(loc);
044                
045                int burnLevel = 25 + burnMod;
046                
047                float speed = Misc.getSpeedForBurnLevel(burnLevel);
048                Vector2f vel = Misc.getUnitVectorAtDegreeAngle(travelDir);
049                vel.scale(speed);
050                setVel(vel);
051                
052                float travelDur = genFloat(15f, 20f);
053                float streamDur = travelDur - (1f + getRandom().nextFloat());
054                
055                setAccelMult(0.1f);
056                //addBehavior(new GBGoInDirection(genDelay(20f), travelDir, burnLevel));
057                //addBehavior(new GBGoInDirection(1000f, travelDir, burnLevel));
058                addBehavior(new GBGoInDirectionWithWobble(travelDur, travelDir,
059                                                        0.01f + 0.04f * manager.getRandom().nextFloat(), 
060                                                        10f + 20f * manager.getRandom().nextFloat(), burnLevel));
061                addInterrupt(new GBIGenerateSlipstream(size, size * 1.2f, burnLevel - 5, 30f, 20, streamDur));
062                
063                
064                
065//              if (!placeNearPlayer()) {
066//                      setCreationFailed();
067//                      return;
068//              }
069//              addBehavior(new GBStayInPlace(10f));
070//              addInterrupt(new GBITooClose(0f, pf, 500f));
071//              addBehavior(new GBSlipstreamPath(genDelay(5f), pf, burnLevel, true));
072//              addInterrupt(new GBIGenerateSlipstream(size, burnLevel - 5, 50f, 20));
073
074        }
075}
