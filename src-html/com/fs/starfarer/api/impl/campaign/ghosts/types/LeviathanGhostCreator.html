001package com.fs.starfarer.api.impl.campaign.ghosts.types;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import com.fs.starfarer.api.Global;
007import com.fs.starfarer.api.impl.campaign.ghosts.BaseSensorGhostCreator;
008import com.fs.starfarer.api.impl.campaign.ghosts.GhostFrequencies;
009import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhost;
010import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhostManager;
011
012public class LeviathanGhostCreator extends BaseSensorGhostCreator {
013
014        @Override
015        public List<SensorGhost> createGhost(SensorGhostManager manager) {
016                if (!Global.getSector().getCurrentLocation().isHyperspace()) return null;
017                
018                float r = manager.getRandom().nextFloat();
019                
020                boolean withCalf = false;
021                int numRemora = 0;
022                if (r < 0.1f) {
023                        withCalf = true;
024                } else if (r < 0.3f) {
025                        numRemora = 1 + manager.getRandom().nextInt(3);
026                }
027                
028//              calf = false;
029//              numRemora = 3;
030                
031                List<SensorGhost> result = new ArrayList<SensorGhost>();
032                LeviathanGhost g = new LeviathanGhost(manager, 0);
033                if (g.isCreationFailed()) return null;
034                
035                result.add(g);
036                
037                if (withCalf) {
038                        LeviathanCalfGhost c = new LeviathanCalfGhost(manager, g.getEntity());
039                        result.add(c);
040                }
041                if (numRemora > 0) {
042                        for (int i = 0; i < numRemora; i++) {
043                                RemoraGhost remora = new RemoraGhost(manager, g.getEntity(), 1000f); // however long the Leviathan lasts
044                                if (!remora.isCreationFailed()) {
045                                        result.add(remora);
046                                }
047                        }
048                }
049                
050                return result;
051        }
052
053        
054        @Override
055        public float getFrequency(SensorGhostManager manager) {
056                //return 100000f;
057                return GhostFrequencies.getLeviathanFrequency(manager);
058        }
059        
060        public boolean canSpawnWhilePlayerInOrNearSlipstream() {
061                return true;
062        }
063}
