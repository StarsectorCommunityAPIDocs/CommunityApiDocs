001package com.fs.starfarer.api.impl.campaign.ghosts.types;
002
003import java.util.Random;
004
005import org.lwjgl.util.vector.Vector2f;
006
007import com.fs.starfarer.api.impl.campaign.ghosts.BaseSensorGhost;
008import com.fs.starfarer.api.impl.campaign.ghosts.GBFollowStream;
009import com.fs.starfarer.api.impl.campaign.ghosts.GBGoInDirection;
010import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhostManager;
011import com.fs.starfarer.api.impl.campaign.velfield.SlipstreamTerrainPlugin2;
012import com.fs.starfarer.api.impl.campaign.velfield.SlipstreamTerrainPlugin2.SlipstreamSegment;
013import com.fs.starfarer.api.util.Misc;
014
015public class RacerGhost extends BaseSensorGhost {
016
017        public RacerGhost(SensorGhostManager manager, SlipstreamSegment segment, SlipstreamTerrainPlugin2 plugin) {
018                super(manager, 20);
019                
020                initEntity(genLargeSensorProfile(), genSmallRadius());
021                setDespawnRange(-entity.getRadius() * 0.5f);
022                
023                Random random = manager.getRandom();
024                float offset = (random.nextBoolean() ? 1f : -1f) * random.nextFloat() * 0.5f;
025                
026                Vector2f loc = plugin.getPointAt(segment.totalLength, offset);
027                if (loc == null) {
028                        setCreationFailed();
029                        return;
030                }
031                setLoc(loc);
032                
033                float intensity = plugin.getIntensity(offset);
034                float wMult = plugin.getWidthBasedSpeedMult(segment.totalLength);
035                float b = plugin.getFaderBrightness(segment.totalLength);
036                float speedMult = intensity * wMult * b;
037                
038                float speed = Misc.getSpeedForBurnLevel(plugin.getParams().burnLevel);
039                Vector2f vel = new Vector2f(segment.dir);
040                vel.scale(speed * speedMult);
041                setVel(vel);
042                
043                int burn = 17 + random.nextInt(15);
044                
045                float dur = 5f + 3f * random.nextFloat();
046                addBehavior(new GBFollowStream(dur, burn, plugin));
047                addBehavior(new GBGoInDirection(dur * 0.5f, random.nextFloat() * 360f, burn));
048        }
049}
