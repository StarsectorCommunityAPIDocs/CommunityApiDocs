001package com.fs.starfarer.api.impl.campaign.ghosts.types;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import com.fs.starfarer.api.Global;
007import com.fs.starfarer.api.impl.campaign.ghosts.BaseSensorGhostCreator;
008import com.fs.starfarer.api.impl.campaign.ghosts.GhostFrequencies;
009import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhost;
010import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhostManager;
011
012public class StormcallerGhostCreator extends BaseSensorGhostCreator {
013
014        @Override
015        public List<SensorGhost> createGhost(SensorGhostManager manager) {
016                if (!Global.getSector().getCurrentLocation().isHyperspace()) return null;
017
018                
019                List<SensorGhost> result = new ArrayList<SensorGhost>();
020                SensorGhost g = new StormcallerGhost(manager);
021                if (g.isCreationFailed()) return null;
022                result.add(g);
023                return result;
024        }
025
026        @Override
027        public float getFrequency(SensorGhostManager manager) {
028                return GhostFrequencies.getStormcallerFrequency(manager);
029                //return 10000f;
030        }
031        
032        public boolean canSpawnWhilePlayerInOrNearSlipstream() {
033                return true;
034        }
035}
