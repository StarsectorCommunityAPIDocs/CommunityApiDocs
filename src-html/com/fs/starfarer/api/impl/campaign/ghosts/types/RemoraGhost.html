001package com.fs.starfarer.api.impl.campaign.ghosts.types;
002
003import org.lwjgl.util.vector.Vector2f;
004
005import com.fs.starfarer.api.campaign.CampaignFleetAPI;
006import com.fs.starfarer.api.campaign.SectorEntityToken;
007import com.fs.starfarer.api.impl.campaign.ghosts.BaseSensorGhost;
008import com.fs.starfarer.api.impl.campaign.ghosts.GBCircle;
009import com.fs.starfarer.api.impl.campaign.ghosts.GBFollowClosely;
010import com.fs.starfarer.api.impl.campaign.ghosts.GBGoAwayFrom;
011import com.fs.starfarer.api.impl.campaign.ghosts.GBIRemoraDrain;
012import com.fs.starfarer.api.impl.campaign.ghosts.GBIntercept;
013import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhostManager;
014import com.fs.starfarer.api.impl.campaign.ids.Tags;
015
016public class RemoraGhost extends BaseSensorGhost {
017
018        public RemoraGhost(SensorGhostManager manager, SectorEntityToken target, float duration) {
019                this(manager, target, duration, null);
020        }
021        public RemoraGhost(SensorGhostManager manager, SectorEntityToken target, float duration, Vector2f loc) {
022                super(manager, 50);
023                
024                float circleRadius = genFloat(0f, 50f);
025                if (target instanceof CampaignFleetAPI) {
026                        circleRadius = genFloat(300f, 500f);
027                }
028                
029                initEntity(genMediumSensorProfile(), genSmallRadius());
030                setDespawnRange(0f);
031                entity.addTag(Tags.IMMUNE_TO_REMORA_PULSE);
032                
033                if (loc != null) {
034                        setLoc(loc);
035                } else if (target.isPlayerFleet()) {
036                        if (!placeNearPlayer()) {
037                                setCreationFailed();
038                                return;
039                        }
040                } else {
041                        placeNearEntity(target, 200f, 300f);
042                }
043                
044                addBehavior(new GBIntercept(target, 10f, 25, circleRadius + 500f, true));
045                if (target instanceof CampaignFleetAPI) {
046                        setAccelMult(5f);
047                        addBehavior(new GBCircle(target, duration, 25 + manager.getRandom().nextInt(6), circleRadius, 1f));
048                } else {
049                        setAccelMult(0.1f); // GBFollowClosely doesn't work well otherwise
050                        addBehavior(new GBFollowClosely(target, duration, 50, 200f, 300f));
051                }
052                addInterrupt(new GBIRemoraDrain(target, 750f));
053                addBehavior(new GBGoAwayFrom(5f, target, 20));
054        }
055}
