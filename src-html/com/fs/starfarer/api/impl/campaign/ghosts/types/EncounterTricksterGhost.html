001package com.fs.starfarer.api.impl.campaign.ghosts.types;
002
003import java.util.Random;
004
005import org.lwjgl.util.vector.Vector2f;
006
007import com.fs.starfarer.api.Global;
008import com.fs.starfarer.api.campaign.CampaignFleetAPI;
009import com.fs.starfarer.api.campaign.FleetAssignment;
010import com.fs.starfarer.api.campaign.SectorEntityToken;
011import com.fs.starfarer.api.impl.campaign.ghosts.BaseSensorGhost;
012import com.fs.starfarer.api.impl.campaign.ghosts.GBCircle;
013import com.fs.starfarer.api.impl.campaign.ghosts.GBITowardsEntity;
014import com.fs.starfarer.api.impl.campaign.ghosts.GBIntercept;
015import com.fs.starfarer.api.impl.campaign.ghosts.GBLeadPlayerTo;
016import com.fs.starfarer.api.impl.campaign.ghosts.SensorGhostManager;
017
018public class EncounterTricksterGhost extends BaseSensorGhost {
019
020        public EncounterTricksterGhost(SensorGhostManager manager, CampaignFleetAPI other, boolean guideToTarget) {
021                super(manager, guideToTarget ? 40 : 20);
022                
023                Random random = manager.getRandom();
024                CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
025                float circleRadius = genFloat(300f, 400f);
026                
027                initEntity(genLargeSensorProfile(), genSmallRadius());
028                placeNearEntity(other, 0f, 0f);
029                
030                setDespawnRange(-1000f);
031                
032                if (guideToTarget) {
033                        SectorEntityToken e = pf.getContainingLocation().createToken(new Vector2f(other.getLocation()));
034                        other.addAssignment(FleetAssignment.ORBIT_AGGRESSIVE,
035                                                                e, 10f + 20f * random.nextFloat(), "investigating anomalous readings");
036                        
037                        addBehavior(new GBIntercept(pf, 10f, 40, circleRadius + 500f, true));
038                        addBehavior(new GBCircle(pf, genDelay(2f), 40, circleRadius, 1f));
039                        addInterrupt(new GBITowardsEntity(genDelay(1.5f), pf, other));
040                        addBehavior(new GBLeadPlayerTo(20f, other, genFloat(500f, 600f), 25));
041                } else {
042                        other.addAssignment(FleetAssignment.ORBIT_AGGRESSIVE,
043                                        entity, 10f + 20f * random.nextFloat(), "investigating sensor ghost");
044                        
045                        addBehavior(new GBIntercept(pf, 10f, (int)Math.round(other.getFleetData().getBurnLevel()) + 1, 100f, true));
046                }
047                
048        }
049        
050}
