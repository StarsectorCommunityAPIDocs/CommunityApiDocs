001package com.fs.starfarer.api.impl.campaign.econ.impl;
002
003import java.awt.Color;
004
005import com.fs.starfarer.api.Global;
006import com.fs.starfarer.api.campaign.PlanetAPI;
007import com.fs.starfarer.api.campaign.econ.CommoditySpecAPI;
008import com.fs.starfarer.api.impl.campaign.ids.Conditions;
009import com.fs.starfarer.api.impl.campaign.ids.Stats;
010import com.fs.starfarer.api.impl.campaign.ids.Strings;
011import com.fs.starfarer.api.ui.TooltipMakerAPI;
012import com.fs.starfarer.api.util.Misc;
013
014
015
016public class PlanetaryShield extends BaseIndustry {
017
018        public static float DEFENSE_BONUS = 2f;
019        
020        public static float ALPHA_CORE_BONUS = 0.5f;
021        public static float IMPROVE_DEFENSE_BONUS = 0.25f;
022        
023        public void apply() {
024                super.apply(false);
025                
026                int size = 5;
027                applyIncomeAndUpkeep(size);
028                
029                float bonus = DEFENSE_BONUS;
030                market.getStats().getDynamic().getMod(Stats.GROUND_DEFENSES_MOD)
031                                                .modifyMult(getModId(), 1f + bonus, getNameForModifier());
032                
033                if (isFunctional()) {
034                        market.suppressCondition(Conditions.METEOR_IMPACTS);
035                        applyVisuals(market.getPlanetEntity());
036                } else {
037                        unapply();
038                }
039        }
040        
041        public static void applyVisuals(PlanetAPI planet) {
042                if (planet == null) return;
043                planet.getSpec().setShieldTexture(Global.getSettings().getSpriteName("industry", "shield_texture"));
044                //planet.getSpec().setShieldThickness(0.07f);
045                //planet.getSpec().setShieldColor(new Color(255,0,0,255));
046                planet.getSpec().setShieldThickness(0.1f);
047                planet.getSpec().setShieldColor(new Color(255,255,255,175));
048                planet.applySpecChanges();
049        }
050        
051        public static void unapplyVisuals(PlanetAPI planet) {
052                if (planet == null) return;
053                planet.getSpec().setShieldTexture(null);
054                planet.getSpec().setShieldThickness(0f);
055                planet.getSpec().setShieldColor(null);
056                planet.applySpecChanges();
057        }
058
059        
060        @Override
061        public void unapply() {
062                super.unapply();
063                
064                market.unsuppressCondition(Conditions.METEOR_IMPACTS);
065                unapplyVisuals(market.getPlanetEntity());
066                
067                market.getStats().getDynamic().getMod(Stats.GROUND_DEFENSES_MOD).unmodifyMult(getModId());
068        }
069
070        @Override
071        public boolean isAvailableToBuild() {
072                if (!Global.getSector().getPlayerFaction().knowsIndustry(getId())) {
073                        return false;
074                }
075                return market.getPlanetEntity() != null && !market.getPlanetEntity().isGasGiant();
076        }
077        
078        @Override
079        public String getUnavailableReason() {
080                if (!super.isAvailableToBuild()) return super.getUnavailableReason();
081                return "Can not be built at a gas giant";
082        }
083        
084        public boolean showWhenUnavailable() {
085                return Global.getSector().getPlayerFaction().knowsIndustry(getId());
086        }
087
088        protected boolean hasPostDemandSection(boolean hasDemand, IndustryTooltipMode mode) {
089                return mode != IndustryTooltipMode.NORMAL || isFunctional();
090        }
091        
092        @Override
093        protected void addPostDemandSection(TooltipMakerAPI tooltip, boolean hasDemand, IndustryTooltipMode mode) {
094                if (mode != IndustryTooltipMode.NORMAL || isFunctional()) {
095                        
096                        float bonus = DEFENSE_BONUS;
097                        addGroundDefensesImpactSection(tooltip, bonus, (String[])null);
098                }
099        }
100        
101        
102        @Override
103        protected void applyAlphaCoreModifiers() {
104                market.getStats().getDynamic().getMod(Stats.GROUND_DEFENSES_MOD).modifyMult(
105                                getModId(1), 1f + ALPHA_CORE_BONUS, "Alpha core (" + getNameForModifier() + ")");
106        }
107        
108        @Override
109        protected void applyNoAICoreModifiers() {
110                market.getStats().getDynamic().getMod(Stats.GROUND_DEFENSES_MOD).unmodifyMult(getModId(1));
111        }
112        
113        @Override
114        protected void applyAlphaCoreSupplyAndDemandModifiers() {
115                demandReduction.modifyFlat(getModId(0), DEMAND_REDUCTION, "Alpha core");
116        }
117        
118        protected void addAlphaCoreDescription(TooltipMakerAPI tooltip, AICoreDescriptionMode mode) {
119                float opad = 10f;
120                Color highlight = Misc.getHighlightColor();
121                
122                String pre = "Alpha-level AI core currently assigned. ";
123                if (mode == AICoreDescriptionMode.MANAGE_CORE_DIALOG_LIST || mode == AICoreDescriptionMode.INDUSTRY_TOOLTIP) {
124                        pre = "Alpha-level AI core. ";
125                }
126                float a = ALPHA_CORE_BONUS;
127                //String str = Strings.X + (int)Math.round(a * 100f) + "%";
128                String str = Strings.X + (1f + a) + "";
129                
130                if (mode == AICoreDescriptionMode.INDUSTRY_TOOLTIP) {
131                        CommoditySpecAPI coreSpec = Global.getSettings().getCommoditySpec(aiCoreId);
132                        TooltipMakerAPI text = tooltip.beginImageWithText(coreSpec.getIconName(), 48);
133                        text.addPara(pre + "Reduces upkeep cost by %s. Reduces demand by %s unit. " +
134                                        "Increases ground defenses by %s.", 0f, highlight,
135                                        "" + (int)((1f - UPKEEP_MULT) * 100f) + "%", "" + DEMAND_REDUCTION,
136                                        str);
137                        tooltip.addImageWithText(opad);
138                        return;
139                }
140                
141                tooltip.addPara(pre + "Reduces upkeep cost by %s. Reduces demand by %s unit. " +
142                                "Increases ground defenses by %s.", opad, highlight,
143                                "" + (int)((1f - UPKEEP_MULT) * 100f) + "%", "" + DEMAND_REDUCTION,
144                                str);
145                
146        }
147
148        
149        @Override
150        public boolean canImprove() {
151                return true;
152        }
153        
154        protected void applyImproveModifiers() {
155                if (isImproved()) {
156                        market.getStats().getDynamic().getMod(Stats.GROUND_DEFENSES_MOD).modifyMult(getModId(2),
157                                                        1f + IMPROVE_DEFENSE_BONUS,
158                                                        getImprovementsDescForModifiers() + " (" + getNameForModifier() + ")");
159                } else {
160                        market.getStats().getDynamic().getMod(Stats.GROUND_DEFENSES_MOD).unmodifyMult(getModId(2));
161                }
162        }
163        
164        public void addImproveDesc(TooltipMakerAPI info, ImprovementDescriptionMode mode) {
165                float opad = 10f;
166                Color highlight = Misc.getHighlightColor();
167                
168                float a = IMPROVE_DEFENSE_BONUS;
169                String str = Strings.X + (1f + a) + "";
170                
171                if (mode == ImprovementDescriptionMode.INDUSTRY_TOOLTIP) {
172                        info.addPara("Ground defenses increased by %s.", 0f, highlight, str);
173                } else {
174                        info.addPara("Increases ground defenses by %s.", 0f, highlight, str);
175                }
176
177                info.addSpacer(opad);
178                super.addImproveDesc(info, mode);
179        }
180}
