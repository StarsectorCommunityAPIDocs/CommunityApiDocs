001package com.fs.starfarer.api.impl.campaign.econ.impl;
002
003import java.util.ArrayList;
004import java.util.List;
005
006
007public class ConstructionQueue {
008
009        public static class ConstructionQueueItem {
010                public String id;
011                public int cost;
012                public ConstructionQueueItem(String id, int cost) {
013                        this.id = id;
014                        this.cost = cost;
015                }
016        }
017        
018        protected List<ConstructionQueueItem> items = new ArrayList<ConstructionQueueItem>();
019        
020        protected Object readResolve() {
021                if (items == null) {
022                        items = new ArrayList<ConstructionQueueItem>();
023                }
024                return this;
025        }
026
027        public List<ConstructionQueueItem> getItems() {
028                return items;
029        }
030        public void setItems(List<ConstructionQueueItem> items) {
031                this.items = items;
032        }
033        
034        public void addToEnd(String id, int cost) {
035                ConstructionQueueItem item = new ConstructionQueueItem(id, cost);
036                items.add(item);
037        }
038        
039        public void moveUp(String id) {
040                ConstructionQueueItem item = getItem(id);
041                if (item == null) return;
042                int index = items.indexOf(item);
043                index--;
044                if (index < 0) index = 0;
045                items.remove(item);
046                items.add(index, item);
047        }
048        
049        public void moveDown(String id) {
050                ConstructionQueueItem item = getItem(id);
051                if (item == null) return;
052                int index = items.indexOf(item);
053                index++;
054                items.remove(item);
055                if (index > items.size()) index = items.size();
056                items.add(index, item);
057        }
058        
059        public void moveToFront(String id) {
060                ConstructionQueueItem item = getItem(id);
061                if (item == null) return;
062                items.remove(item);
063                items.add(0, item);
064        }
065        
066        public void moveToBack(String id) {
067                ConstructionQueueItem item = getItem(id);
068                if (item == null) return;
069                items.remove(item);
070                items.add(item);
071        }
072        
073        public void removeItem(String id) {
074                items.remove(getItem(id));
075        }
076        
077        public ConstructionQueueItem getItem(String id) {
078                for (ConstructionQueueItem item : items) {
079                        if (item.id.equals(id)) return item;
080                }
081                return null;
082        }
083        
084        public boolean hasItem(String id) {
085                return getItem(id) != null;
086        }
087}
