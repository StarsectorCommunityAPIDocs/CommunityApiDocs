010public class FuelProduction extends BaseIndustry {
011
012        public void apply() {
013                super.apply(true);
014                supplyBonus.modifyFlat(getModId(2), market.getAdmin().getStats().getDynamic().getValue(Stats.FUEL_SUPPLY_BONUS_MOD, 0), "Administrator");
015                
016                int size = market.getSize();
017                
018                demand(Commodities.VOLATILES, size);
019                demand(Commodities.HEAVY_MACHINERY, size - 2);
020                
021                supply(Commodities.FUEL, size - 2);
022                
023                Pair<String, Integer> deficit = getMaxDeficit(Commodities.VOLATILES);
024                
025                applyDeficitToProduction(1, deficit, Commodities.FUEL);
026                
027                if (!isFunctional()) {
028                        supply.clear();
029                }
030        }
031
032        
033        @Override
034        public void unapply() {
035                super.unapply();
036        }
037        
038
039        @Override
040        public String getCurrentImage() {
041                if (getSpecialItem() != null) {
042                        return Global.getSettings().getSpriteName("industry", "advanced_fuel_prod");
043                }
044                return super.getCurrentImage();
045        }
046
047
048        public boolean isDemandLegal(CommodityOnMarketAPI com) {
049                return true;
050        }
051
052        public boolean isSupplyLegal(CommodityOnMarketAPI com) {
053                return true;
054        }
