067        @Override
068        public boolean isAvailableToBuild() {
069                if (!super.isAvailableToBuild()) return false;
070                boolean aquaculture = Industries.AQUACULTURE.equals(getId());
071                boolean canAquaculture = market.getPlanetEntity() != null && 
072                                                                 AQUA_PLANETS.contains(market.getPlanetEntity().getTypeId());
073                if (aquaculture != canAquaculture) return false;
074                
075                for (MarketConditionAPI mc : market.getConditions()) {
076                        String commodity = ResourceDepositsCondition.COMMODITY.get(mc.getId());
077                        if (commodity != null) {
078                                String industry = ResourceDepositsCondition.INDUSTRY.get(commodity);
079                                if (Industries.FARMING.equals(industry)) return true;
080                        }
081                }
082                return false;
083        }

154        @Override
155        public String getCurrentImage() {
156                boolean aquaculture = Industries.AQUACULTURE.equals(getId());
157                if (aquaculture) {
158                        return super.getCurrentImage();
159                }
160                float size = market.getSize();
161                //PlanetAPI planet = market.getPlanetEntity();
162                if (size <= SIZE_FOR_SMALL_IMAGE) {
163                        return Global.getSettings().getSpriteName("industry", "farming_low");
164                }
165                if (size >= SIZE_FOR_LARGE_IMAGE) {
166                        return Global.getSettings().getSpriteName("industry", "farming_high");
167                }
168                return Global.getSettings().getSpriteName("industry", "farming_med");
169                //return super.getCurrentImage();
170        }
