018public class HeavyIndustry extends BaseIndustry {
019
020        public static float ORBITAL_WORKS_QUALITY_BONUS = 0.2f;
021        
022        public static String POLLUTION_ID = Conditions.POLLUTION;
023        public static float DAYS_BEFORE_POLLUTION = 0f;
024        public static float DAYS_BEFORE_POLLUTION_PERMANENT = 90f;
025        
026        
027        public void apply() {
028                super.apply(true);
029                
030                int size = market.getSize();
031                
032                boolean works = Industries.ORBITALWORKS.equals(getId());
033                int shipBonus = 0;
034                float qualityBonus = 0;
035                if (works) {
036                        //shipBonus = 2;
037                        qualityBonus = ORBITAL_WORKS_QUALITY_BONUS;
038                }
039                
040                demand(Commodities.METALS, size);
041                demand(Commodities.RARE_METALS, size - 2);
042                
043                supply(Commodities.HEAVY_MACHINERY, size - 2);
044                supply(Commodities.SUPPLIES, size - 2);
045                supply(Commodities.HAND_WEAPONS, size - 2);
046                supply(Commodities.SHIPS, size - 2);
047                if (shipBonus > 0) {
048                        supply(1, Commodities.SHIPS, shipBonus, "Orbital works");
049                }
050                
051                Pair<String, Integer> deficit = getMaxDeficit(Commodities.METALS, Commodities.RARE_METALS);
052                int maxDeficit = size - 3; // to allow *some* production so economy doesn't get into an unrecoverable state
053                if (deficit.two > maxDeficit) deficit.two = maxDeficit;
054                
055                applyDeficitToProduction(2, deficit,
056                                        Commodities.HEAVY_MACHINERY,
057                                        Commodities.SUPPLIES,
058                                        Commodities.HAND_WEAPONS,
059                                        Commodities.SHIPS);
060                
061//              if (market.getId().equals("chicomoztoc")) {
062//                      System.out.println("efwefwe");
063//              }
064                
065                if (qualityBonus > 0) {
066                        market.getStats().getDynamic().getMod(Stats.PRODUCTION_QUALITY_MOD).modifyFlat(getModId(1), qualityBonus, "Orbital works");
067                }
068                
069                float stability = market.getPrevStability();
070                if (stability < 5) {
071                        float stabilityMod = (stability - 5f) / 5f;
072                        stabilityMod *= 0.5f;
073                        //market.getStats().getDynamic().getMod(Stats.PRODUCTION_QUALITY_MOD).modifyFlat(getModId(0), stabilityMod, "Low stability at production source");
074                        market.getStats().getDynamic().getMod(Stats.PRODUCTION_QUALITY_MOD).modifyFlat(getModId(0), stabilityMod, getNameForModifier() + " - low stability");
075                }
076                
077                if (!isFunctional()) {
078                        supply.clear();
079                        unapply();
080                }
081        }
082        
083        @Override
084        public void unapply() {
085                super.unapply();
086                
087                market.getStats().getDynamic().getMod(Stats.PRODUCTION_QUALITY_MOD).unmodifyFlat(getModId(0));
088                market.getStats().getDynamic().getMod(Stats.PRODUCTION_QUALITY_MOD).unmodifyFlat(getModId(1));
089        }
090
091        
092        @Override
093        protected void addPostDemandSection(TooltipMakerAPI tooltip, boolean hasDemand, IndustryTooltipMode mode) {
094                //if (mode == IndustryTooltipMode.NORMAL && isFunctional()) {
095                if (mode != IndustryTooltipMode.NORMAL || isFunctional()) {                     
096                        boolean works = Industries.ORBITALWORKS.equals(getId());
097                        if (works) {
098                                float total = ORBITAL_WORKS_QUALITY_BONUS;
099                                String totalStr = "+" + (int)Math.round(total * 100f) + "%";
100                                Color h = Misc.getHighlightColor();
101                                if (total < 0) {
102                                        h = Misc.getNegativeHighlightColor();
103                                        totalStr = "" + (int)Math.round(total * 100f) + "%";
104                                }
105                                float opad = 10f;
106                                if (total >= 0) {
107                                        tooltip.addPara("Ship quality: %s", opad, h, totalStr);
108                                        tooltip.addPara("*Quality bonus only applies for the largest ship producer in the faction.", 
109                                                        Misc.getGrayColor(), opad);
110                                }
111                        }
112                }
113        }

128        @Override
129        public boolean wantsToUseSpecialItem(SpecialItemData data) {
130                if (special != null && Items.CORRUPTED_NANOFORGE.equals(special.getId()) &&
131                                data != null && Items.PRISTINE_NANOFORGE.equals(data.getId())) {
132                        return true;
133                }

135        }
136
137        protected boolean permaPollution = false;
138        protected boolean addedPollution = false;
139        protected float daysWithNanoforge = 0f;
140        
141        @Override
142        public void advance(float amount) {
143                super.advance(amount);
144                
145                if (special != null) {
146                        float days = Global.getSector().getClock().convertToDays(amount);
147                        daysWithNanoforge += days;
148        
149                        updatePollutionStatus();
150                }
151        }
152        
153        protected void updatePollutionStatus() {
154                if (!market.hasCondition(Conditions.HABITABLE)) return;
155                
156                if (special != null) {
157                        if (!addedPollution && daysWithNanoforge >= DAYS_BEFORE_POLLUTION) {
158                                if (market.hasCondition(POLLUTION_ID)) {
159                                        permaPollution = true;
160                                } else {
161                                        market.addCondition(POLLUTION_ID);
162                                        addedPollution = true;
163                                }
164                        }
165                        if (addedPollution && !permaPollution) {
166                                if (daysWithNanoforge > DAYS_BEFORE_POLLUTION_PERMANENT) {
167                                        permaPollution = true;
168                                }
169                        }
170                } else if (addedPollution && !permaPollution) {
171                        market.removeCondition(POLLUTION_ID);
172                        addedPollution = false;
173                }
174        }
175

182        }

186        }
187
188        public void setAddedPollution(boolean addedPollution) {
189                this.addedPollution = addedPollution;
190        }
191
192        public float getDaysWithNanoforge() {
193                return daysWithNanoforge;
194        }
195
196        public void setDaysWithNanoforge(float daysWithNanoforge) {
197                this.daysWithNanoforge = daysWithNanoforge;
198        }
199
200        @Override
201        public void setSpecialItem(SpecialItemData special) {
202                super.setSpecialItem(special);
203                
204                updatePollutionStatus();
205        }
