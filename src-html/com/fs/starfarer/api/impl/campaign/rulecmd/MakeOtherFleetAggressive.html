001package com.fs.starfarer.api.impl.campaign.rulecmd;
002
003import java.util.List;
004import java.util.Map;
005
006import com.fs.starfarer.api.campaign.CampaignFleetAPI;
007import com.fs.starfarer.api.campaign.InteractionDialogAPI;
008import com.fs.starfarer.api.campaign.ai.ModularFleetAIAPI;
009import com.fs.starfarer.api.campaign.rules.MemoryAPI;
010import com.fs.starfarer.api.impl.campaign.ids.MemFlags;
011import com.fs.starfarer.api.util.Misc;
012import com.fs.starfarer.api.util.Misc.Token;
013
014/**
015 * Doesn't matter unless the other fleet is also hostile or preventing disengage.
016 * @author Alex Mosolov
017 *
018 * Copyright 2016 Fractal Softworks, LLC
019 */
020public class MakeOtherFleetAggressive extends BaseCommandPlugin {
021
022        public boolean execute(String ruleId, InteractionDialogAPI dialog, List<Token> params, Map<String, MemoryAPI> memoryMap) {
023
024                if (dialog == null) return false;
025                if (!(dialog.getInteractionTarget() instanceof CampaignFleetAPI)) return false;
026                
027                String reason = "generic";
028                boolean value;
029                float expire = 0f;
030                if (params.size() >= 2) {
031                        reason = params.get(0).getString(memoryMap);
032                        value = params.get(1).getBoolean(memoryMap);
033                        if (params.size() >= 3) {
034                                expire = params.get(2).getFloat(memoryMap);
035                        }
036                } else if (params.size() < 1) {
037                        value = true;
038                } else {
039                        value = params.get(0).getBoolean(memoryMap);
040                }
041                
042                MemoryAPI memory = dialog.getInteractionTarget().getMemoryWithoutUpdate();
043                Misc.setFlagWithReason(memory, MemFlags.MEMORY_KEY_MAKE_AGGRESSIVE, reason, value, expire);
044                
045                CampaignFleetAPI fleet = (CampaignFleetAPI) dialog.getInteractionTarget();
046                if (fleet.getAI() instanceof ModularFleetAIAPI) {
047                        ModularFleetAIAPI mAI = (ModularFleetAIAPI) fleet.getAI();
048                        mAI.getTacticalModule().forceTargetReEval();
049                }
050
051                return true;
052        }
053
054}
