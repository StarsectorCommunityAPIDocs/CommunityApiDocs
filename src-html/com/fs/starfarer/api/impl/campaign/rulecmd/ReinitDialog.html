012/**
013 * SetActiveMission $missionEventHandle
014 */
015public class ReinitDialog extends BaseCommandPlugin {
016        public boolean execute(String ruleId, InteractionDialogAPI dialog, List<Token> params, Map<String, MemoryAPI> memoryMap) {
017                if (!(dialog.getPlugin() instanceof RuleBasedDialog)) return false;
018                
019                ((RuleBasedDialog) dialog.getPlugin()).reinit(true);
020                
021                return true;
022        }
023
024}
