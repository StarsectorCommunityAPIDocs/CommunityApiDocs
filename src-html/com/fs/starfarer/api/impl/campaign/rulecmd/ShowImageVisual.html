003import java.util.List;
004import java.util.Map;
005
006import com.fs.starfarer.api.Global;

008import com.fs.starfarer.api.campaign.CampaignFleetAPI;

015
016/**
017 * ShowImageVisual <category> <key>
018 */
019public class ShowImageVisual extends BaseCommandPlugin {
020
021        public ShowImageVisual() {
022                
023        }
024        
025        public boolean execute(String ruleId, InteractionDialogAPI dialog, List<Token> params, Map<String, MemoryAPI> memoryMap) {
026                
027                
028                String category = "illustrations";
029                String key = null;
030                
031                if (params.size() <= 1) {
032                        key = params.get(0).string;
033                } else {
034                        category = params.get(0).string;
035                        key = params.get(1).string;
036                }
037                
038                SpriteAPI sprite = Global.getSettings().getSprite(category, key);
039                dialog.getVisualPanel().showImagePortion(category, key, sprite.getWidth(), sprite.getHeight(), 0, 0, 480, 300);
040                
041                return true;
042        }
043        
044        protected void showFleetInfo(InteractionDialogAPI dialog, CampaignFleetAPI player, CampaignFleetAPI other) {
045                BattleAPI b = player.getBattle();
046                if (b == null) b = other.getBattle();
047                if (b != null && b.isPlayerInvolved()) {
048                        String titleOne = "Your forces";
049                        if (b.isPlayerInvolved() && b.getPlayerSide().size() > 1) {
050                                titleOne += ", with allies";
051                        }
052                        if (!Global.getSector().getPlayerFleet().isValidPlayerFleet()) {
053                                titleOne = "Allied forces";
054                        }
055                        String titleTwo = null;
056                        if (b.getPrimary(b.getNonPlayerSide()) != null) {
057                                titleTwo = b.getPrimary(b.getNonPlayerSide()).getNameWithFactionKeepCase();
058                        }
059                        if (b.getNonPlayerSide().size() > 1) titleTwo += ", with allies";
060                        dialog.getVisualPanel().showFleetInfo(titleOne, b.getPlayerCombined(), Misc.ucFirst(titleTwo), b.getNonPlayerCombined(), null);
061                } else {
062                        if (b != null) {
063                                String titleOne = b.getPrimary(b.getSideOne()).getNameWithFactionKeepCase();
064                                if (b.getSideOne().size() > 1) titleOne += ", with allies";
065                                String titleTwo = b.getPrimary(b.getSideTwo()).getNameWithFactionKeepCase();
066                                if (b.getSideTwo().size() > 1) titleTwo += ", with allies";
067                                
068                                FleetEncounterContext fake = new FleetEncounterContext();
069                                fake.setBattle(b);
070                                dialog.getVisualPanel().showPreBattleJoinInfo(null, player, Misc.ucFirst(titleOne), Misc.ucFirst(titleTwo), fake);
071                        } else {
072                                dialog.getVisualPanel().showFleetInfo((String)null, player, (String)null, other, null);
073                        }
074                }
075        }
076
077}
