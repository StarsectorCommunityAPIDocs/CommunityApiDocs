001package com.fs.starfarer.api.impl.campaign.tutorial;
002
003import com.fs.starfarer.api.Global;
004import com.fs.starfarer.api.campaign.econ.MarketAPI;
005import com.fs.starfarer.api.campaign.econ.MonthlyReport;
006import com.fs.starfarer.api.campaign.econ.MonthlyReport.FDNode;
007import com.fs.starfarer.api.campaign.listeners.EconomyTickListener;
008import com.fs.starfarer.api.impl.campaign.shared.SharedData;
009import com.fs.starfarer.api.ui.TooltipMakerAPI;
010import com.fs.starfarer.api.ui.TooltipMakerAPI.TooltipCreator;
011import com.fs.starfarer.api.util.Misc;
012
013public class GalatianAcademyStipend implements EconomyTickListener, TooltipCreator {
014
015        public static float DURATION = 365 + 365 + 365 + 20;
016        public static int STIPEND = 15000;
017        
018        protected long startTime = 0;
019        public GalatianAcademyStipend() {
020                Global.getSector().getListenerManager().addListener(this);
021                startTime = Global.getSector().getClock().getTimestamp();
022                Global.getSector().getMemoryWithoutUpdate().set("$playerReceivingGAStipend", true);
023        }
024        
025        public void reportEconomyTick(int iterIndex) {
026                if (!Global.getSettings().getBoolean("enableStipend")) return;
027                
028                int lastIterInMonth = (int) Global.getSettings().getFloat("economyIterPerMonth") - 1;
029                if (iterIndex != lastIterInMonth) return;
030                
031                float daysActive = Global.getSector().getClock().getElapsedDaysSince(startTime);
032                MarketAPI ancyra = Global.getSector().getEconomy().getMarket("ancyra_market");
033                if (daysActive > DURATION || ancyra == null) {
034                        Global.getSector().getListenerManager().removeListener(this);
035                        Global.getSector().getMemoryWithoutUpdate().unset("$playerReceivingGAStipend");
036                        return;
037                }
038                
039                
040                
041                MonthlyReport report = SharedData.getData().getCurrentReport();
042
043
044                int stipend = getStipend();
045                FDNode fleetNode = report.getNode(MonthlyReport.FLEET);
046                
047                FDNode stipendNode = report.getNode(fleetNode, "GA_stipend");
048                stipendNode.income = stipend;
049                stipendNode.name = "Stipend from Galatia Academy";
050                stipendNode.icon = Global.getSettings().getSpriteName("income_report", "generic_income");
051                stipendNode.tooltipCreator = this;
052        }
053        
054        protected int getStipend() {
055                return STIPEND;
056        }
057
058        public void reportEconomyMonthEnd() {
059        }
060
061        public void createTooltip(TooltipMakerAPI tooltip, boolean expanded, Object tooltipParam) {
062                float daysActive = Global.getSector().getClock().getElapsedDaysSince(startTime);
063                tooltip.addPara("A monthly stipend of %s from the Galatian Academy, " +
064                                "for the aid you've recently rendered.", 
065                                0f, Misc.getHighlightColor(), Misc.getDGSCredits(getStipend()));
066                
067                float rem = DURATION - daysActive;
068                int months = (int) (rem / 30f);
069                //if (months > 0 && months <= 12) {
070                if (months > 0) {
071                        tooltip.addPara("You should continue receiving the stipend for another %s months.", 10f,
072                                        Misc.getHighlightColor(), "" + months);
073                } else if (months <= 0) {
074                        tooltip.addPara("This month's payment was the last.", 10f);
075                }
076        }
077
078        public float getTooltipWidth(Object tooltipParam) {
079                return 450;
080        }
081
082        public boolean isTooltipExpandable(Object tooltipParam) {
083                return false;
084        }
085}
