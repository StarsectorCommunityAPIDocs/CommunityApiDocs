019public interface WeaponAPI {
020        
021        public static interface DerivedWeaponStatsAPI {
022                float getBurstFireDuration();
023                float getSustainedDps();
024                float getEmpPerSecond();
025                float getDamageOver30Sec();
026                float getDps();
027                float getBurstDamage();
028                float getFluxPerDam();
029                float getRoF();
030                float getFluxPerSecond();
031                float getSustainedFluxPerSecond();
032                /**
033                 * Multiplied by number of barrels for LINKED, by 2 for DUAL_LINKED, and by the number of missiles for MIRVs.
034                 * @return
035                 */
036                float getDamagePerShot();
037                float getEmpPerShot();
038        }
039        
040        public static enum WeaponType {
041                BALLISTIC("Ballistic"),
042                ENERGY("Energy"),
043                MISSILE("Missile"),
044                LAUNCH_BAY("Launch Bay"),
045                UNIVERSAL("Universal"),
046                HYBRID("Hybrid"),
047                SYNERGY("Synergy"),
048                COMPOSITE("Composite"),
049                BUILT_IN("Built in"),
050                DECORATIVE("Decorative"),
051                SYSTEM("System"),
052                STATION_MODULE("Station Module");
053                
054                private String displayName;
055                private WeaponType(String displayName) {
056                        this.displayName = displayName;
057                }
058                public String getDisplayName() {
059                        return displayName;
060                }
061                
062        }
063        
064        public static enum WeaponSize {
065                SMALL("Small"),
066                MEDIUM("Medium"),
067                LARGE("Large");
068                
069                private String displayName;
070                private WeaponSize(String name) {
071                        this.displayName = name;
072                }
073                public String getDisplayName() {
074                        return displayName;
075                }
076        }
077        
078        public static enum AIHints {
079                PD,
080                PD_ONLY,
081                PD_ALSO, // PD, but only if there are no other targets
082                USE_VS_FRIGATES,
083                STRIKE,
084                DANGEROUS, /** like STRIKE but only for when the enemy ship is considering it, not for its own weapon use */
085                BOMB,
086                GUIDED_POOR,
087                DO_NOT_AIM,
088                ANTI_FTR,
089                HEATSEEKER,
090                SYSTEM,
091                SHOW_IN_CODEX,
092                AUTOZOOM,
093                DO_NOT_CONSERVE,
094                CONSERVE_1,
095                CONSERVE_2,
096                CONSERVE_3,
097                CONSERVE_4,
098                CONSERVE_5,
099                CONSERVE_ALL,
100                CONSERVE_FOR_ANTI_ARMOR,
101                FIRE_WHEN_INEFFICIENT,
102                EXTRA_RANGE_ON_FIGHTER,
103                
104                IGNORES_FLARES,
105                
106                GROUP_LINKED,
107                GROUP_ALTERNATING,
108                
109                MISSILE_SPREAD,
110                DIRECT_AIM,
111                NO_TURN_RATE_BOOST_WHEN_IDLE,
112                RESET_BARREL_INDEX_ON_BURST,
113                
114                USE_LESS_VS_SHIELDS,
115        }

180        DamageType getDamageType();
181        
182        ShipAPI getShip();
183        
184        /**
185         * Base stats, does not include character skill bonuses/hull mods/etc.
186         * @return
187         */
188        DerivedWeaponStatsAPI getDerivedStats();
189        
190        void setAmmo(int ammo);
191
192        /**
193         * @return null for non-animated weapons.
194         */
195        AnimationAPI getAnimation();
196        
197        
198        /**
199         * Note:
200         * setAlphaMult() and setAngle() will be called on the sprite returned here just prior to rendering.
201         * Thus, setting these is pointless - the values will be overridden. Uses the alpha channel in SpriteAPI.setColor()

203         * 
204         * @return either the base sprite, or, for animated weapons, the sprite for the current frame.
205         */
206        SpriteAPI getSprite();

209         * "Base" sprite for the weapon (see: mjolnir.wpn), or null.

211         */

213        
214        /**
215         * Sprite with the weapon barrels, or null if the weapon doesn't use recoil/separate barrel graphics.
216         * @return
217         */
218        SpriteAPI getBarrelSpriteAPI();
219        /**
220         * Renders the barrel. Shouldn't need to do this unless for shaders etc.
221         * @param alphaMult
222         */
223        void renderBarrel(SpriteAPI sprite, Vector2f loc, float alphaMult);

225         * Whether the barrel goes below or above the weapon sprite.
226         * @return
227         */

272        WeaponSpecAPI getOriginalSpec();
273        void setWeaponGlowWidthMult(float weaponGlowWidthMult);
274        void setWeaponGlowHeightMult(float weaponGlowHeightMult);
