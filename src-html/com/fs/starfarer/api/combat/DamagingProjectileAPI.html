001package com.fs.starfarer.api.combat;
002
003import java.util.List;
004
005import org.lwjgl.util.vector.Vector2f;
006
007import com.fs.starfarer.api.loading.ProjectileSpawnType;
008import com.fs.starfarer.api.loading.ProjectileSpecAPI;
009
010public interface DamagingProjectileAPI extends CombatEntityAPI {
011        
012        DamageType getDamageType();
013        float getDamageAmount();
014        
015        /**
016         * Does not include damage reduction from projectile fading out or having done damage
017         * @return
018         */
019        float getBaseDamageAmount();
020        float getEmpAmount();
021        
022        
023        void setDamageAmount(float damage);
024        
025        
026        /**
027         * @return Weapon that fired this projectile. Can be null (for example, if spawned without one via the API).
028         */
029        WeaponAPI getWeapon();
030
031        /**
032         * Whether the projectile already did its damage and is now fading out.
033         * @return
034         */
035        boolean didDamage();
036        
037        /**
038         * @return What the damage was dealt to, once didDamage() returns true. Can be null.
039         */
040        CombatEntityAPI getDamageTarget();
041        
042        String getProjectileSpecId();
043        
044        
045        /**
046         * Generally a ShipAPI for the ship that ultimately fired this weapon. Can be null.
047         * 
048         * Projectiles can't hit their source, except for fizzled-out missiles.
049         * 
050         * @return
051         */
052        ShipAPI getSource();
053        void setSource(ShipAPI source);
054        
055        /**
056         * @return whether the projectile has started fading out due to exceeding its maximum range.
057         */
058        boolean isFading();
059        
060        ProjectileSpawnType getSpawnType();
061        
062        
063        /**
064         * Time the projectile has been alive.
065         * @return
066         */
067        float getElapsed();
068        
069        
070        DamageAPI getDamage();
071        
072        boolean isFromMissile();
073        
074        /**
075         * Should be set to true for BALLISTIC, BALLISTIC_AS_BEAM, and PLASMA_SHOT projectiles 
076         * spawned from a missile.
077         * Needed for incoming damage evaluation AI to function properly in these cases.
078         * @param fromMissile
079         */
080        void setFromMissile(boolean fromMissile);
081        
082        
083        /**
084         * Only supported by damaging explosions, not other types of projectiles.
085         * @param c
086         */
087        void removeDamagedAlready(CombatEntityAPI c);
088        
089        /**
090         * Only supported by damaging explosions, not other types of projectiles.
091         * @param c
092         */
093        void addDamagedAlready(CombatEntityAPI c);
094        float getMoveSpeed();
095        Vector2f getSpawnLocation();
096        ProjectileSpecAPI getProjectileSpec();
097        float getBrightness();
098        
099        /**
100         * Only non-null for "moving ray" and "ballistic projectile" type projectiles, not missiles/plasma shots/etc.
101         * @return
102         */
103        Vector2f getTailEnd();
104        List<CombatEntityAPI> getDamagedAlready();
105        
106}
