021        public static List<String> modifyDamageDealt(Object param, CombatEntityAPI target, 
022                                                                                         DamageAPI damage, Vector2f point, boolean shieldHit) {
023                return modifyDamageDealt(Global.getCombatEngine().getListenerManager().getListeners(DamageDealtModifier.class),
024                                                                 param, target, damage, point, shieldHit);
025        }
026        
027        public static List<String> modifyDamageDealt(ShipAPI attacker, Object param, CombatEntityAPI target, 
028                                                                                                 DamageAPI damage, Vector2f point, boolean shieldHit) {
029                if (attacker == null || attacker.getListenerManager() == null) return null;
030                return modifyDamageDealt(attacker.getListeners(DamageDealtModifier.class), param, target, damage, point, shieldHit);
031        }
032        
033        protected static List<String> modifyDamageDealt(List<DamageDealtModifier> mods, Object param, CombatEntityAPI target, 
034                                                                                                 DamageAPI damage, Vector2f point, boolean shieldHit) {
035                List<String> ids = null;
036                mods = new ArrayList<DamageDealtModifier>(mods);
037                for (DamageDealtModifier x : mods) {
038                        String id = x.modifyDamageDealt(param, target, damage, point, shieldHit);
039                        if (id != null) {
040                                if (ids == null) ids = new ArrayList<String>();
041                                ids.add(id);
042                        }
043                }
044                return ids;
045        }

058                // ship == target, yes
059                if (ship == null || ship.getListenerManager() == null) return null;

077        public static void reportDamageApplied(ShipAPI ship, Object source, CombatEntityAPI target, ApplyDamageResultAPI result) {
078                for (DamageListener x : new ArrayList<DamageListener>(Global.getCombatEngine().getListenerManager().getListeners(DamageListener.class))) {
079                        x.reportDamageApplied(source, target, result);
080                }
081                if (ship != null) {
082                        for (DamageListener x : new ArrayList<DamageListener>(ship.getListeners(DamageListener.class))) {
083                                x.reportDamageApplied(source, target, result);
084                        }
085                }
086        }

088        public static boolean notifyAboutToTakeHullDamage(ShipAPI ship, Object source, Vector2f point, float damageAmount) {
089                boolean result = false;
090                for (HullDamageAboutToBeTakenListener x : new ArrayList<HullDamageAboutToBeTakenListener>(Global.getCombatEngine().getListenerManager().getListeners(HullDamageAboutToBeTakenListener.class))) {
091                        result |= x.notifyAboutToTakeHullDamage(source, ship, point, damageAmount);
092                }
093                for (HullDamageAboutToBeTakenListener x : new ArrayList<HullDamageAboutToBeTakenListener>(ship.getListeners(HullDamageAboutToBeTakenListener.class))) {
094                        result |= x.notifyAboutToTakeHullDamage(source, ship, point, damageAmount);
095                }
096                return result;
097        }

099        public static float getWeaponRangePercentMod(ShipAPI ship, WeaponAPI weapon) {
100                float mod = 0f;
101                if (ship != null) {
102                        for (WeaponRangeModifier x : new ArrayList<WeaponRangeModifier>(ship.getListeners(WeaponRangeModifier.class))) {
103                                mod += x.getWeaponRangePercentMod(ship, weapon);
104                        }
105                }
106                return mod;
107        }

113                                mod *= x.getWeaponRangeMultMod(ship, weapon);
114                        }
