001package com.fs.starfarer.api.combat.listeners;
002
003import org.lwjgl.util.vector.Vector2f;
004
005import com.fs.starfarer.api.combat.CombatEntityAPI;
006import com.fs.starfarer.api.combat.DamageAPI;
007
008/**
009 * When adding to a ship, add to the ship dealing the damage, NOT the one taking the damage.
010 * 
011 * Should add this to a ship where possible instead of the engine as performance will be better.
012 * 
013 * @author Alex Mosolov
014 *
015 * Copyright 2019 Fractal Softworks, LLC
016 */
017public interface DamageDealtModifier {
018        
019        /**
020         * Passed in as param to modifyDamageDealt/Taken when the damage dealer is an EMP ship system,
021         * which does not create EmpArcEntityAPI's for historical reasons. 
022         */
023        public static String EMP_SHIP_SYSTEM_PARAM = "EMP_SHIP_SYSTEM_PARAM";
024        
025        /**
026         * Modifications to damage should ONLY be made using damage.getModifier().
027         * 
028         * param can be:
029         * null
030         * DamagingProjectileAPI
031         * BeamAPI
032         * EmpArcEntityAPI
033         * Something custom set by a script
034         * 
035         * @return the id of the stat modification to damage.getModifier(), or null if no modification was made
036         */
037        String modifyDamageDealt(Object param, CombatEntityAPI target, 
038                                                DamageAPI damage, Vector2f point, boolean shieldHit);
039        
040}
