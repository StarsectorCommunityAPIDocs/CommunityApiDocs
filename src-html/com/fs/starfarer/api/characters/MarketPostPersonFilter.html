001package com.fs.starfarer.api.characters;
002
003import com.fs.starfarer.api.Global;
004import com.fs.starfarer.api.campaign.econ.MarketAPI;
005import com.fs.starfarer.api.characters.ImportantPeopleAPI.PersonDataAPI;
006import com.fs.starfarer.api.characters.ImportantPeopleAPI.PersonFilter;
007
008public class MarketPostPersonFilter implements PersonFilter {
009
010        private final String marketId;
011        private final String [] postIds;
012        private final String checkoutReason;
013
014        public MarketPostPersonFilter(String marketId, String checkoutReason, String ... postIds) {
015                this.marketId = marketId;
016                this.checkoutReason = checkoutReason;
017                this.postIds = postIds;
018        }
019        
020        public boolean accept(PersonDataAPI personData) {
021                if (!Global.getSector().getImportantPeople().canCheckOutPerson(personData.getPerson(), checkoutReason)) {
022                        return false;
023                }
024                
025                if (postIds != null) {
026                        boolean postMatchFound = false;
027                        for (String id : postIds) {
028                                if (id.equals(personData.getPerson().getPostId())) {
029                                        postMatchFound = true;
030                                }
031                        }
032                        if (!postMatchFound) return false;
033                }
034                
035                MarketAPI market = personData.getLocation().getMarket();
036                if (market == null || !market.getId().equals(marketId)) return false;
037                return true;
038        }
039
040}
